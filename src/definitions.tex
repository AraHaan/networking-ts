\rSec0[defs]{Terms and definitions}

\setcounter{section}{1}
\setcounter{tocdepth}{0}

\indexdefn{host byte order}%
\definition{host byte order}{defs.host.byte.order}
see section 3.194 of POSIX Base Definitions, Host Byte Order

\indexdefn{network byte order}%
\definition{network byte order}{defs.net.byte.order}
see section 3.238 of POSIX Base Definitions, Network Byte Order

\indexdefn{synchronous operation}%
\definition{synchronous operation}{defs.sync.op}
an operation where control is not returned until the operation completes

\indexdefn{asynchronous operation}%
\definition{asynchronous operation}{defs.async.op}
an operation where control is returned immediately without waiting for the operation to complete
\begin{note} Multiple asynchronous operations may be executed concurrently. \end{note}

\indexdefn{orderly shutdown}%
\definition{orderly shutdown}{defs.orderly.shutdown}
the procedure for shutting down a stream after all work in progress has been completed, without loss of data

\rSec1[convenience.hdr.synop]{Header \tcode{<experimental/net>} synopsis}

\begin{codeblock}
#include <experimental/executor>

#include <experimental/buffer>
#include <experimental/socket>
#include <experimental/internet>
\end{codeblock}

\pnum
\enternote This header is provided as a convenience for programs so that they may access all networking facilities via a single, self-contained \tcode{\#include}. \exitnote
